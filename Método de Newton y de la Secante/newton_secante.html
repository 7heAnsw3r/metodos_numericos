    return -3 * x**2 + np.sin(x)

def metodo_newton(f,f_prime, p0, tol=1e-10,max_iter=100):
    for i in range(max_iter):
        f_p0 = f(p0)
        f_prime_p0 = f_prime(p0)
        if abs(f_prime_p0) < tol:
            print('La derivada es muy cercana a cero, el m√©todo no converge.')
            return None
        p1 = p0 - f_p0 / f_prime_p0
        print(f'Iteraci√≥n {i+1}: p{i} = {p0}, p{i+1} = {p1}')
        if abs(p1 - p0) < tol:  # Converge si la diferencia es menor a la tolerancia
            return p1
        p0 = p1
    print('No se alcanz√≥ la tolerancia en el n√∫mero m√°ximo de iteraciones.')
    return p0

def metodo_secante(f, p0, p1, tol=1e-10, max_iter=100):
    for i in range(max_iter):
        f_p0 = f(p0)
        f_p1 = f(p1)
        if abs(f_p1 - f_p0) < tol:  # Evitar divisi√≥n por cero
            print('La diferencia entre f(p0) y f(p1) es muy peque√±a, el m√©todo no converge.')
            return None
        p2 = p1 - f_p1 * (p1 - p0) / (f_p1 - f_p0)
        print(f'Iteraci√≥n {i+1}: p{i} = {p0}, p{i+1} = {p1}, p{i+2} = {p2}')
        if abs(p2 - p1) < tol:  # Converge si la diferencia es menor a la tolerancia
            return p2
        p0, p1 = p1, p2
    print('No se alcanz√≥ la tolerancia en el n√∫mero m√°ximo de iteraciones.')
    return p1

            return p

            b = p
